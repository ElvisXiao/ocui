Example
var tree = new Tree({...})
tree.addNode = function(model, cb){
		$.ajax({
			url: '/add',
			type: 'put',
			data: model,
			success: function(){
				cb(true);
			},
			error: function(res){
				oc.dialog.tips(res.responseText);
