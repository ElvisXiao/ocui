用于操作浏览器的本地存储 - LocalStorage
Source:

Example

//设置一个键值对到本地存储中
oc.localStorage.set('user', {name: 'elvis xiao', email: 'ivesxiao@gmail.com'});

//根据主键获取存储的值
oc.localStorage.get('user');

//删除指定key
oc.localStorage.remove('user');

//清除所有该域下本地存储
oc.localStorage.clear();
